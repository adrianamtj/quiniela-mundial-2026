function calcPts(pL,pV,rL,rV){

  if(rL==null||rV==null) return null;

  if(pL===rL&&pV===rV) return 2;

  const pr=pL>pV?'L':pL<pV?'V':'E', rr=rL>rV?'L':rL<rV?'V':'E';

  return pr===rr?1:0;

}

function calcTotal(name){

  let t=0;

  state.partidos.forEach(p=>{ if(p.resultLocal==null) return; const pr=(state.pronosticos[p.id]||{})[name]; if(pr) t+=calcPts(pr.local,pr.visita,p.resultLocal,p.resultVisita)||0; });

  return t;

}
